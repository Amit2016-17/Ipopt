275 template<class T>
276 bool MittelmannParaCntrlBase<T>::InitializeProblem(
277  Index N
278  )
279 {
280  typename T::ProblemSpecs p;
281 
282  if( N < 1 )
283  {
284  printf("N has to be at least 1.");
285  return false;
286  }
287 
288  T_ = p.T();
289  l_ = p.l();
290  Nt_ = N;
291  Nx_ = N;
292  dt_ = T_ / Nt_;
293  dx_ = l_ / Nx_;
294  lb_y_ = p.lb_y();
295  ub_y_ = p.ub_y();
296  lb_u_ = p.lb_u();
297  ub_u_ = p.ub_u();
298  alpha_ = p.alpha();
299  beta_ = p.beta();
300 
301  y_T_ = new Number[Nx_ + 1];
302  for( Index j = 0; j <= Nx_; j++ )
303  {
304  y_T_[j] = p.y_T(x_grid(j));
305  }
306  a_y_ = new Number[Nt_];
307  for( Index i = 1; i <= Nt_; i++ )
308  {
309  a_y_[i - 1] = p.a_y(t_grid(i));
310  }
311  a_u_ = new Number[Nt_];
312  for( Index i = 1; i <= Nt_; i++ )
313  {
314  a_u_[i - 1] = p.a_u(t_grid(i));
315  }
316 
317  return true;
318 }

463 template<class T>
464 bool MittelmannParaCntrlBase<T>::eval_f(
465  Index n,
466  const Number* x,
467  bool new_x,
468  Number& obj_value
469  )
470 {
471  // Deviation of y from target
472  Number a = x[y_index(0, Nt_)] - y_T_[0];
473  Number sum = 0.5 * a * a;
474  for( Index jx = 1; jx < Nx_; jx++ )
475  {
476  a = x[y_index(jx, Nt_)] - y_T_[jx];
477  sum += a * a;
478  }
479  a = x[y_index(Nx_, Nt_)] - y_T_[Nx_];
480  sum += 0.5 * a * a;
481  obj_value = 0.5 * dx_ * sum;
482 
483  // smoothing for control
484  if( alpha_ != .0 )
485  {
486  sum = 0.5 * x[u_index(Nt_)] * x[u_index(Nt_)];
487  for( Index it = 1; it < Nt_; it++ )
488  {
489  sum += x[u_index(it)] * x[u_index(it)];
490  }
491  obj_value += 0.5 * alpha_ * dt_ * sum;
492  }
493 
494  // third term
495  sum = 0.;
496  for( Index it = 1; it < Nt_; it++ )
497  {
498  sum += a_y_[it - 1] * x[y_index(Nx_, it)] + a_u_[it - 1] * x[u_index(it)];
499  }
500  sum += 0.5 * (a_y_[Nt_ - 1] * x[y_index(Nx_, Nt_)] + a_u_[Nt_ - 1] * x[u_index(Nt_)]);
501  obj_value += dt_ * sum;
502 
503  return true;
504 }

506 template<class T>
507 bool MittelmannParaCntrlBase<T>::eval_grad_f(
508  Index n,
509  const Number* x,
510  bool new_x,
511  Number* grad_f
512  )
513 {
514  // First set all y entries to zero
515  for( Index jx = 0; jx <= Nx_; jx++ )
516  {
517  for( Index it = 0; it <= Nt_; it++ )
518  {
519  grad_f[y_index(jx, it)] = 0.;
520  }
521  }
522 
523  // y entries from y target
524  grad_f[y_index(0, Nt_)] = 0.5 * dx_ * (x[y_index(0, Nt_)] - y_T_[0]);
525  for( Index jx = 1; jx < Nx_; jx++ )
526  {
527  grad_f[y_index(jx, Nt_)] = dx_ * (x[y_index(jx, Nt_)] - y_T_[jx]);
528  }
529  grad_f[y_index(Nx_, Nt_)] = 0.5 * dx_ * (x[y_index(Nx_, Nt_)] - y_T_[Nx_]);
530 
531  // y entries from thrid term
532  for( Index it = 1; it < Nt_; it++ )
533  {
534  grad_f[y_index(Nx_, it)] = dt_ * a_y_[it - 1];
535  }
536  grad_f[y_index(Nx_, Nt_)] += 0.5 * dt_ * a_y_[Nt_ - 1];
537 
538  // u entries from smoothing and third term
539  for( Index it = 1; it < Nt_; it++ )
540  {
541  grad_f[u_index(it)] = alpha_ * dt_ * x[u_index(it)] + dt_ * a_u_[it - 1];
542  }
543  grad_f[u_index(Nt_)] = 0.5 * dt_ * (alpha_ * x[u_index(Nt_)] + a_u_[Nt_ - 1]);
544 
545  return true;
546 }

548 template<class T>
549 bool MittelmannParaCntrlBase<T>::eval_g(
550  Index n,
551  const Number* x,
552  bool new_x,
553  Index m,
554  Number* g
555  )
556 {
557  typename T::ProblemSpecs p;
558 
559  Index ig = 0;
560 
561  Number f = 1. / (2. * dx_ * dx_);
562  for( Index jx = 1; jx < Nx_; jx++ )
563  {
564  for( Index it = 0; it < Nt_; it++ )
565  {
566  g[ig] = (x[y_index(jx, it)] - x[y_index(jx, it + 1)]) / dt_
567  + f
568  * (x[y_index(jx - 1, it)] - 2. * x[y_index(jx, it)] + x[y_index(jx + 1, it)] + x[y_index(jx - 1, it + 1)]
569  - 2. * x[y_index(jx, it + 1)] + x[y_index(jx + 1, it + 1)]);
570  ig++;
571  }
572  }
573 
574  for( Index it = 1; it <= Nt_; it++ )
575  {
576  g[ig] = (x[y_index(2, it)] - 4. * x[y_index(1, it)] + 3. * x[y_index(0, it)]);
577  ig++;
578  }
579 
580  f = 1. / (2. * dx_);
581  for( Index it = 1; it <= Nt_; it++ )
582  {
583  g[ig] = f * (x[y_index(Nx_ - 2, it)] - 4. * x[y_index(Nx_ - 1, it)] + 3. * x[y_index(Nx_, it)])
584  + beta_ * x[y_index(Nx_, it)] - x[u_index(it)] + p.phi(x[y_index(Nx_, it)]);
585  ig++;
586  }
587 
588  DBG_ASSERT(ig == m);
589 
590  return true;
591 }

710 template<class T>
711 bool MittelmannParaCntrlBase<T>::eval_h(
712  Index n,
713  const Number* x,
714  bool new_x,
715  Number obj_factor,
716  Index m,
717  const Number* lambda,
718  bool new_lambda,
719  Index nele_hess,
720  Index* iRow,
721  Index* jCol,
722  Number* values
723  )
724 {
725  typename T::ProblemSpecs p;
726 
727  Index ihes = 0;
728 
729  if( values == NULL )
730  {
731  // y values from objective
732  for( Index jx = 0; jx <= Nx_; jx++ )
733  {
734  iRow[ihes] = y_index(jx, Nt_);
735  jCol[ihes] = y_index(jx, Nt_);
736  ihes++;
737  }
738  // u from objective
739  for( Index it = 1; it <= Nt_; it++ )
740  {
741  iRow[ihes] = u_index(it);
742  jCol[ihes] = u_index(it);
743  ihes++;
744  }
745 
746  // constraint
747  if( !p.phi_dydy_always_zero() )
748  {
749  for( Index it = 1; it <= Nt_; it++ )
750  {
751  iRow[ihes] = y_index(Nx_, it);
752  jCol[ihes] = y_index(Nx_, it);
753  ihes++;
754  }
755  }
756  }
757  else
758  {
759  // y values from objective
760  values[ihes++] = obj_factor * 0.5 * dx_;
761  for( Index jx = 1; jx < Nx_; jx++ )
762  {
763  values[ihes++] = obj_factor * dx_;
764  }
765  values[ihes++] = obj_factor * 0.5 * dx_;
766  // u from objective
767  for( Index it = 1; it < Nt_; it++ )
768  {
769  values[ihes++] = obj_factor * alpha_ * dt_;
770  }
771  values[ihes++] = obj_factor * 0.5 * alpha_ * dt_;
772 
773  // constrainT
774  if( !p.phi_dydy_always_zero() )
775  {
776  Index ig = (Nx_ - 1) * Nt_ + Nt_;
777  for( Index it = 1; it <= Nt_; it++ )
778  {
779  values[ihes++] = lambda[ig++] * p.phi_dydy(x[y_index(Nx_, it)]);
780  }
781  }
782  }
783 
784  DBG_ASSERT(ihes==nele_hess);
785 
786  return true;
787 }
