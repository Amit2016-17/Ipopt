7 // This file is part of the Ipopt tutorial. It is a version with
8 // mistakes for the C++ implementation of the coding exercise problem
9 // (in AMPL formulation):
10 //
11 // param n := 4;
12 //
13 // var x {1..n} <= 0, >= -1.5, := -0.5;
14 //
15 // minimize obj:
16 // sum{i in 1..n} (x[i]-1)^2;
17 //
18 // subject to constr {i in 2..n-1}:
19 // (x[i]^2+1.5*x[i]-i/n)*cos(x[i+1]) - x[i-1] = 0;
20 //
21 // The constant term "i/n" in the constraint is supposed to be input data
22 //
