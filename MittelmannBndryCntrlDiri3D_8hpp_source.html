304  inline Number PenObj(
305  Number t
306  ) const
307  {
308  //return 0.5*t*t;
309  if( t > B_ )
310  {
311  return B_ * B_ / 2. + C_ * (t - B_);
312  }
313  else if( t < -B_ )
314  {
315  return B_ * B_ / 2. + C_ * (-t - B_);
316  }
317  else
318  {
319  const Number t2 = t * t;
320  const Number t4 = t2 * t2;
321  const Number t6 = t4 * t2;
322  return PenA_ * t2 + PenB_ * t4 + PenC_ * t6;
323  }
324  }
