306  inline Number PenObj(
307  Number t
308  ) const
309  {
310  if( B_ == 0. )
311  {
312  return 0.5 * t * t;
313  }
314  else if( t > B_ )
315  {
316  return B_ * B_ / 2. + C_ * (t - B_);
317  }
318  else if( t < -B_ )
319  {
320  return B_ * B_ / 2. + C_ * (-t - B_);
321  }
322  else
323  {
324  const Number t2 = t * t;
325  const Number t4 = t2 * t2;
326  const Number t6 = t4 * t2;
327  return PenA_ * t2 + PenB_ * t4 + PenC_ * t6;
328  }
329  }
330 
332  inline Number PenObj_1(
333  Number t
334  ) const
335  {
336  if( B_ == 0. )
337  {
338  return t;
339  }
340  else if( t > B_ )
341  {
342  return C_;
343  }
344  else if( t < -B_ )
345  {
346  return -C_;
347  }
348  else
349  {
350  const Number t2 = t * t;
351  const Number t3 = t * t2;
352  const Number t5 = t3 * t2;
353  return 2. * PenA_ * t + 4. * PenB_ * t3 + 6. * PenC_ * t5;
354  }
355  }
356 
358  inline Number PenObj_2(
359  Number t
360  ) const

363  {
364  return 1.;
365  }
366  else if( t > B_ )

369  }

377  const Number t4 = t2 * t2;
378  return 2. * PenA_ + 12. * PenB_ * t2 + 30. * PenC_ * t4;
379  }
