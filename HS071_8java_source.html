214  protected boolean eval_h(
215  int n,
216  double[] x,
217  boolean new_x,
218  double obj_factor,
219  int m,
220  double[] lambda,
221  boolean new_lambda,
222  int nele_hess,
223  int[] iRow,
224  int[] jCol,
225  double[] values)
226  {
227  assert n == this.n;
228  assert m == this.m;
229 
230  int idx = 0; /* nonzero element counter */
231  int row = 0; /* row counter for loop */
232  int col = 0; /* col counter for loop */
233 
234  if (values == null) {
235  /* return the structure
236  * This is a symmetric matrix, fill the lower left triangle only.
237  */
238 
239  /* the hessian for this problem is actually dense */
240  idx = 0;
241  for( row = 0; row < n; ++row )
242  for (col = 0; col <= row; ++col)
243  {
244  iRow[idx] = row;
245  jCol[idx] = col;
246  ++idx;
247  }
248 
249  assert idx == nele_hess;
250  assert nele_hess == this.nele_hess;
251  }
252  else
253  {
254  /* return the values.
255  * This is a symmetric matrix, fill the lower left triangle only.
256  */
257 
258  /* fill the objective portion */
259  values[0] = obj_factor * (2 * x[3]); /* 0,0 */
260  values[1] = obj_factor * (x[3]); /* 1,0 */
261  values[2] = 0.0; /* 1,1 */
262  values[3] = obj_factor * (x[3]); /* 2,0 */
263  values[4] = 0.0; /* 2,1 */
264  values[5] = 0.0; /* 2,2 */
265  values[6] = obj_factor * (2 * x[0] + x[1] + x[2]); /* 3,0 */
266  values[7] = obj_factor * (x[0]); /* 3,1 */
267  values[8] = obj_factor * (x[0]); /* 3,2 */
268  values[9] = 0.0; /* 3,3 */
269 
270  /* add the portion for the first constraint */
271  values[1] += lambda[0] * (x[2] * x[3]); /* 1,0 */
272  values[3] += lambda[0] * (x[1] * x[3]); /* 2,0 */
273  values[4] += lambda[0] * (x[0] * x[3]); /* 2,1 */
274  values[6] += lambda[0] * (x[1] * x[2]); /* 3,0 */
275  values[7] += lambda[0] * (x[0] * x[2]); /* 3,1 */
276  values[8] += lambda[0] * (x[0] * x[1]); /* 3,2 */
277 
278  /* add the portion for the second constraint */
279  values[0] += lambda[1] * 2.0; /* 0,0 */
280  values[2] += lambda[1] * 2.0; /* 1,1 */
281  values[5] += lambda[1] * 2.0; /* 2,2 */
282  values[9] += lambda[1] * 2.0; /* 3,3 */
283  }
