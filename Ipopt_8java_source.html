291  public boolean create(
292  int n,
293  int m,
294  int nele_jac,
295  int nele_hess,
296  int index_style)
297  {
298  // delete any previously created native memory
299  dispose();
300 
301  x = new double[n];
302  g = new double[m];
303 
304  // allocate the callback arguments
305  callback_grad_f = new double[n];
306  callback_jac_g = new double[nele_jac];
307  callback_hess = new double[nele_hess];
308 
309  // the multiplier
310  mult_x_U = new double[n];
311  mult_x_L = new double[n];
312  mult_g = new double[m];
313 
314  // create the optimization problem and return a pointer to it
315  ipopt = CreateIpoptProblem(n, m, nele_jac, nele_hess, index_style);
316 
317  //System.out.println("Finish Java Obj");
318  return ipopt == 0 ? false : true;
319  }
